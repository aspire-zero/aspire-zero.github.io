import java.util.*;

public class Main
{
    static int T;
    static int[] q = new int[100];
    public static void get_bit(int a)
    {
        int t = 0;
        while(a > 0)
        {
            if(a % 2 == 1)
                q[t] ++;
            a >>= 1;
            t ++;
        }
        return;
    }
    public static void main(String[] args) {
        Scanner inScanner = new Scanner(System.in);
        T = inScanner.nextInt();
        while(T --> 0)
        {
            int n;
            int sum = 0;
            n = inScanner.nextInt();
            Arrays.fill(q, 0);
            for(int i = 1; i <= n; i ++)
            {
                int a = inScanner.nextInt();
                get_bit(a);
                sum ^= a;
            }
            if(sum == 0)
                System.out.println(0);
            else
            {
                int cnt = -1;
                int u = sum;
                while(u > 0)
                {
                    cnt ++;
                    u >>= 1;
                }
                if(q[cnt] == 1 || (q[cnt] % 2 == 1 && n % 2 == 1))
                    System.out.println(1);
                else
                    System.out.println(-1);
            }
        }
        inScanner.close();
    }
}
