import java.util.*;


public class Main {
    static int N = 100010;
    static int n, m;
    static int[] pre = new int[110];
    public static int find(int k)
    {
        if(pre[k] != -1)
        {    pre[k] = find(pre[k]);
            return pre[k];
        }
        return k;
    }
    public static void main(String[] args) {
        Scanner inScanner = new Scanner(System.in);
        n = inScanner.nextInt();
        Arrays.fill(pre, -1);
        int ans = 0;
        for(int i = 1; i < n; i ++)
        {
            int x, y;
            x = inScanner.nextInt();
            y = inScanner.nextInt();
            pre[x] = y;
        }
        inScanner.close();
        for(int i = 1; i <= n; i ++)
        {
            if(pre[i] == -1)
            {
                ans = i;
                break;
            }
        }
        for(int i = 1; i <= n; i ++)
        {
            if(i != ans && find(i) != ans)
            {
                ans = -1;
                break;
            }
        }
        System.out.println(ans);
    }
}
